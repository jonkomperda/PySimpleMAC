323  double precision, parameter :: conv = 0.00010d0

332  iter = iter + 1
333  change = 0.0d0
334 
335  do i=2,nx-1
336  do j=2,ny-1
337  pold=p(i,j)
338  p(i,j) = 0.250d0*((p(i-1,j)+p(i,j-1)+p(i+1,j)+p(i,j+1))-(q(i,j)*dx*dx))
339  p(i,j) = pold + rf*(p(i,j)-pold);
340

343 
344  !Stores greatest change value
